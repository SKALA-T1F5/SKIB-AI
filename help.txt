# conda 가상환경 생성

conda activate question-bank

# 텍스트 파일에 있는 모든 패키지 한 번에 설치
pip install -r requirements.txt

# gemini key 사용 패키지 설치

  pytest tests/agents/test_document_analyzer.py -v

  pytest tests/agents/test_question_generator.py -v
